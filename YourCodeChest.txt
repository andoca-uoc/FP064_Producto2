        //Modal ACtualizad Acto
        function abrirModalActualizar(actoJson) {
            var acto = typeof actoJson === 'string' ? JSON.parse(actoJson) : actoJson;

            document.querySelector('#update_fecha').value = acto.date || '';
            document.querySelector('#update_hora').value = acto.time || '';
            document.querySelector('#update_titulo').value = acto.title || '';
            document.querySelector('#update_descripcion_corta').value = acto.description1 || '';
            document.querySelector('#update_descripcion_larga').value = acto.description2 || '';
            document.querySelector('#update_num_asistentes').value = acto.audience || '';
            document.querySelector('#update_id_tipo_acto').value = acto.id_type || '';

            var modal = new bootstrap.Modal(document.getElementById('modalActualizar'));
            modal.show();
        }
        //Modal Acto Eliminar:
        function confirmarEliminacion(url) {
            document.getElementById('btnEliminar').onclick = function() {
                window.location.href = url;
            };
            var modal = new bootstrap.Modal(document.getElementById('confirmarEliminacionModal'));
            modal.show();
        }
        function abrirModalActualizarTipo(tipoActo) {
    // Asegúrate de que estos selectores coincidan con los IDs de los campos en tu modal
    var modal = document.getElementById('modalActualizarTipoActo');

    // Actualizar los valores de los campos del formulario
    modal.querySelector('input[name="id_tipo_acto"]').value = tipoActo.id_tipo_acto;
    modal.querySelector('input[name="descripcion"]').value = tipoActo.descripcion;

    // Mostrar el modal
    var modalInstance = new bootstrap.Modal(modal);
    modalInstance.show();
}
