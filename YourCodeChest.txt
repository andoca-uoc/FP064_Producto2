        function abrirModalActualizar(id, fecha, hora, titulo, descripcion_corta, descripcion_larga, num_asistentes, id_tipo_acto) {            
            document.querySelector('#update_id').value = id;
            document.querySelector('#update_fecha').value = fecha;
            document.querySelector('#update_hora').value = hora;
            document.querySelector('#update_titulo').value = titulo;
            document.querySelector('#update_descripcion_corta').value = descripcion_corta;
            document.querySelector('#update_descripcion_larga').value = descripcion_larga;
            document.querySelector('#update_num_asistentes').value = num_asistentes;
            document.querySelector('#update_id_tipo_acto').value = id_tipo_acto;

            var modal = new bootstrap.Modal(document.getElementById('modalActualizar'));
            modal.show();
        }
